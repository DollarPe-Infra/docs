        ```javascript
        const crypto = require('crypto');

        // Generate authentication headers
        const generateHeaders = (apiKey, apiSecret, body = {}) => {
          const timestamp = Math.floor(Date.now() / 1000);
          const sortedBody = JSON.stringify(body, Object.keys(body).sort());
          const message = `${apiKey}|${timestamp}|${sortedBody}`;
          
          const signature = crypto
            .createHmac('sha256', apiSecret)
            .update(message)
            .digest('base64');
          
          return {
            'Content-Type': 'application/json',
            'X-API-KEY': apiKey,
            'X-TIMESTAMP': timestamp.toString(),
            'X-SIGNATURE': signature
          };
        };

        // Example: Fetch exchange rate
        async function fetchRate() {
          const apiKey = 'YOUR_API_KEY';
          const apiSecret = 'YOUR_API_SECRET';
          
          const requestBody = {
            asset: 'USDT',
            network: 'BSC',
            amount: 100
          };
          
          const headers = generateHeaders(apiKey, apiSecret, requestBody);
          
          const response = await fetch('https://sandbox.dollarpe.xyz/cms/api/v1/payout/fetch-rate', {
            method: 'POST',
            headers,
            body: JSON.stringify(requestBody)
          });
          
          const data = await response.json();
          console.log(data);
        }
        ```

      <Tab title="Python">
        ```python
        import time
        import json
        import hmac
        import base64
        import hashlib
        import requests

        # Generate authentication headers
        def generate_headers(api_key, api_secret, body={}):
            timestamp = int(time.time())
            sorted_body = json.dumps(body, sort_keys=True)
            message = f"{api_key}|{timestamp}|{sorted_body}"
            
            signature = base64.b64encode(
                hmac.new(
                    api_secret.encode('utf-8'),
                    message.encode('utf-8'),
                    hashlib.sha256
                ).digest()
            ).decode('utf-8')
            
            return {
                'Content-Type': 'application/json',
                'X-API-KEY': api_key,
                'X-TIMESTAMP': str(timestamp),
                'X-SIGNATURE': signature
            }

        # Example: Fetch exchange rate
        def fetch_rate():
            api_key = 'YOUR_API_KEY'
            api_secret = 'YOUR_API_SECRET'
            
            request_body = {
                'asset': 'USDT',
                'network': 'BSC',
                'amount': 100
            }
            
            headers = generate_headers(api_key, api_secret, request_body)
            
            response = requests.post(
                'https://sandbox.dollarpe.xyz/pos/api/v1/payout/fetch-rate',
                headers=headers,
                json=request_body
            )
            
            data = response.json()
            print(data)
        ```
